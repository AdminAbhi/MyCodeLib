long long maxPrime(long long n) 
{ 
    if (n <= 3)  return n; 
  
    if (n%2 == 0){
        return maxPrime(n/2);
    } 
    if( n%3 == 0){
        return maxPrime(n/3);
    } 
    for(long long i=5; i*i<=n; i=i+6){
        if (n%i == 0) 
            return maxPrime(n/i);
        if(n%(i+2) == 0)
            return maxPrime(n/(i+2)); 
    } 
    return n; 
}
