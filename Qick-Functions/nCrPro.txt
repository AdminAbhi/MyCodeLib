
long long nCr(int n, int r){
	int MOD7 = 1000000007;
        dp.resize(n, vector<long long> (n, 0));
        dp[0][0] = 1;
        for(int i=1; i<n; i++){
            for(int j=0; j<=i; j++){
                if(j == 0 || j == i) dp[i][j] = 1;
                else dp[i][j] = (dp[i-1][j-1] + dp[i-1][j]) % MOD7;
            }
        }
	return dp[n][r];
}
