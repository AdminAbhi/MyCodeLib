int binarySearch(int[] a, int n, int item)
 {  
    int low = 0;
    int high = n - 1;
    while(low<=high){
    {
      int mid = (low + high)/2;
      if (a[mid] == item) 
	return mid;
      else if (a[mid] < item) 
	low = mid + 1;
      else 
	high = mid - 1;
    } 
    return -1;
 }
